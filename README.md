
How to run nextjs-dashboard app.

cd nextjs-dashboard

npm install

npm run dev


#################################################################
#								#
#	npm run dev						#
#		: Starts the development server.		#
#								#
#	npm run build						#
#		: Builds the app for production.		#
#								#
#	npm start						#
#		: Runs the built app in production mode.	#
#								#
#################################################################
